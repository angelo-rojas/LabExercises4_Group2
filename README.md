# Integer numbers
a = 10
b = 20

print("INTEGER OPERATIONS")
print("Addition:", a + b)
print("Subtraction:", a - b)
print("Multiplication:", a * b)
print("Division:", a / b)
print("Modulus:", a % b)
print("Exponent:", a ** b)

# Float numbers
x = 10.5
y = 2.5

print("\nFLOAT OPERATIONS")
print("Addition:", x + y)
print("Subtraction:", x - y)
print("Multiplication:", x * y)
print("Division:", x / y)
print("Modulus:", x % y)
print("Exponent:", x ** y)

# Complex numbers
c1 = 2 + 3j
c2 = 1 + 4j

print("\nCOMPLEX OPERATIONS")
print("Addition:", c1 + c2)
print("Subtraction:", c1 - c2)
print("Multiplication:", c1 * c2)
print("Division:", c1 / c2)
print("Exponent:", c1 ** 2)
# Modulus not allowed for complex numbers

num1 = 1000000
num2 = 1_000_000
print (num1)
print (num2)

# Creating variables
num_int = 10
num_float = 5.75
num_complex = 3 + 4j

# Printing values
print("Integer value:", num_int)
print("Float value:", num_float)
print("Complex value:", num_complex)

# Checking the type of each variable
print("Type of num_int:", type(num_int))
print("Type of num_float:", type(num_float))
print("Type of num_complex:", type(num_complex))
